country_event = {
	id = wsgranking.101
	is_triggered_only = yes
	hide_window = yes
	trigger = {
		NAND = {
			has_country_flag = wsgranking101
			has_country_flag = wsgranking102
			has_country_flag = wsgranking103
			has_country_flag = wsgranking104
		}
	}
	immediate = {
		if = {
			limit = { NOT = { has_country_flag = wsgranking101 }}
			if = {
				limit = { resource_income_compare = { resource = energy value >= 200 }}
				set_country_flag = wsgranking101

				if = {
					limit = { NOT = { has_country_flag = ban_notice_wsgranking }}

country_event = {
	id = wsgranking.310
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		FROM = { 
			is_finished_megastructure = yes 
			owner = { is_same_value = ROOT }
		}
		is_wg_empire = yes
		check_variable = { which = built_ms_count value < 16 }
	}

	immediate = {
		change_variable = { which = built_ms_count value = 1 }
		if = {
			limit = { NOT = { has_country_flag = wsgranking311 }}
			if = {
				limit = { check_variable = { which = built_ms_count value >= 1 }}
				set_country_flag = wsgranking311
				change_variable = { which = rankpts value = 2 }
				if = {
					limit = { NOT = { has_country_flag = ban_notice_wsgranking }}
					country_event = { id = wsgranking.10311 }
				}
			}
			break = yes
		}
