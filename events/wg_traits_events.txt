# 延后触发移除buff，在月结之前移除
event = {
	id = wg_traits.6
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		every_playable_country = {
			limit = { is_wg_empire = yes }
			if = {
				limit = { has_authority = auth_shenhai }
				country_event = { id = sh_traits.5 days = 27 }
			} else = {
				country_event = { id = wg_traits.5 days = 27 }
			}
		}
	}
}

# S飞特质，没有深海能源时自动移除buff
country_event = {
	id = wg_traits.5
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = {
				OR = {
					has_deficit = sr_se
					NOT = { has_trait = wsg_trait_galatic_engineer }
				}
				has_modifier = wsg_trait_galatic_engineer_buff
			}
			remove_modifier = wsg_trait_galatic_engineer_buff
		}
		if = {
			limit = {
				OR = {
					has_deficit = sr_se
					NOT = { has_trait = wsg_trait_knowledgeable2 }
				}
				has_modifier = wg_modifier_knowledgeable2
			}
			remove_modifier = wg_modifier_knowledgeable2
		}
		if = {
			limit = {
				OR = {
					has_deficit = sr_se
					NOT = { has_trait = wsg_trait_knowledgeable2 }
				}
				has_modifier = wg_modifier_knowledgeable2
			}
			remove_modifier = wg_modifier_knowledgeable2
		}
	}
}
