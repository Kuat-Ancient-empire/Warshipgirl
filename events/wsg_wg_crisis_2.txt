country_event = {
	id = wg_crisis_2.2001
	hide_window = yes
	is_triggered_only = yes
	trigger = {
		NOT = { has_global_flag = all_portal_2_destroyed }
	}
	immediate = {
		initialize_crisis_system_select = yes
		random_system = {
			limit = {
				NOR = {
					has_star_flag = forbit_spawn_system
					has_star_flag = has_wg_crisis_2_megastructure
					is_star_class = sc_wsg_uf
				}
			}
			save_global_event_target_as = wg_invasion_2_false_uf_galaxy_gun_system
			every_fleet_in_system = {
				destroy_fleet = this
			}
			random_system_planet = {
				create_wg_crisis_2_guard_fleet = yes
			}
			create_starbase = {
				size = wg_crisis_starbase
				owner = event_target:wg_crisis_2_country
				effect = {
					while = {
						count = 5
						create_wg_crisis_platform = yes
					}
				}
			}
			set_star_flag = has_wg_crisis_2_megastructure
			set_star_class = sc_black_hole
			star? = {
				every_country = {
					create_point_of_interest = {
						id = wg_invasion_2_false_uf_galaxy_gun_poi.1
						name = "wg_invasion_2_false_uf_galaxy_gun_poi"
						desc = "wg_invasion_2_false_uf_galaxy_gun_poi_desc"
						event_chain = "wg_invasion_chain"
						location = event_target:wg_invasion_2_false_uf_galaxy_gun_system
					}
				}
				set_planet_size = 1
				create_fleet = {
					name = "NAME_wg_crisis_2_false_uf_galaxy_gun"
					effect = {
						set_fleet_flag = wg_crisis_2_false_uf_galaxy_gun
						save_global_event_target_as = wg_crisis_2_false_uf_galaxy_gun
						set_owner = event_target:wg_crisis_2_country
						create_ship = {
							name = "NAME_wg_crisis_2_false_uf_galaxy_gun"
							design = "NAME_wg_crisis_2_false_uf_galaxy_gun"
							prefix = no
						}
						set_location = {
							target = prev
							distance = -1
							angle = 0
						}
						every_country = {
							limit = {
								is_ai = no
							}
							country_event = {
								id = wg_crisis_2.2002
							}
						}
					}
					settings = { 
						can_upgrade = no 
						uses_naval_capacity = no
						can_change_composition = no

					}
