	id = wg_crisis_2.3
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		NOR = { 
			has_global_flag = wg_crisis_disabled
			has_global_flag = wg_invasion_2_happened
			has_global_flag = wg_invasion_2_end
		}
	}

	immediate = {
		set_update_modifiers_batch = begin
		if = {
			limit = { has_global_flag = activated_uf_hidden_diff }

		set_global_flag = resistant_army_will_defence

		every_playable_country = {
			limit = { is_ai = no }
			country_event = { id = wg_crisis.75 days = 20 }
		}
		random_system = {
			limit = {
				NOR = { 
					has_star_flag = forbit_spawn_system
					is_star_class = sc_black_hole 
					any_system_planet = {
						is_planet_class = pc_black_hole
					}
				}
			}
			set_star_class = sc_wsg_uf
			set_star_flag = first_portal_2_system
			set_star_flag = cannot_be_hit_by_aow_starfall_spear
			save_global_event_target_as = first_portal_2_system
			random_system_planet = {
				initial_uf_crisis_2_country = yes
				event_target:wg_crisis_2_country = {
					create_fleet = {
						name = "NAME_Dimensional_Portal"
						effect = {
							set_owner = event_target:wg_crisis_2_country
							create_portal = yes
							set_location = {
								target = PREVPREV
								distance = 40
								angle = random
							}
							save_global_event_target_as = first_dimensional_portal_2
							set_fleet_flag = first_dimensional_portal_2
							set_event_locked = yes
							# 简单难度开门后再刷2队弱的
							# 普通难度开门后再刷4队弱的
							# 困难难度开门后再刷4队正常的
							# 萌新难度开门后再刷8队正常的
							# 天灾难度开门后再刷10队正常的
							fleet_event = { id = wg_crisis_2.4 days = 180 } 
							fleet_event = { id = wg_crisis_2.4 days = 360 } 

							if = {
								# 困难以上难度开门后额外刷4队舰队
								limit = {
									OR = {
										has_global_flag = wg_crisis_normal
										has_global_flag = wg_crisis_hard
										has_global_flag = wg_crisis_insane
									}
								}
								fleet_event = { id = wg_crisis_2.4 days = 540 } 

								create_wg_crisis_2_greater_fleet = yes
								last_created_fleet = {
									set_name = "NAME_auxiliary_guard_fleet"
									set_timed_fleet_flag = { flag = defence_fleet days = 4500 }
									set_formation_scale = 2
									fleet_event = { id = wg_crisis_2.401 days = 4501 }
								}
							}
							
							if = {
								# 困难以上难度开门后额外刷2队舰队
								limit = {
									OR = {
										has_global_flag = wg_crisis_hard
										has_global_flag = wg_crisis_insane
									}
								}
								fleet_event = { id = wg_crisis_2.4 days = 720 } 
								fleet_event = { id = wg_crisis.4 days = 900 } 
								fleet_event = { id = wg_crisis_2.102 days = 10800 } #循环刷辅助守家舰队
							}

							if = {
								# 天灾难度开门后再刷2+1+1队
								limit = { has_global_flag = wg_crisis_insane }

								fleet_event = { id = wg_crisis_2.4 days = 1080 } 
								fleet_event = { id = wg_crisis_2.4 days = 1260 } 

								create_wg_crisis_2_greater_fleet = yes

									set_name = "NAME_auxiliary_guard_fleet"
									set_timed_fleet_flag = { flag = defence_fleet days = 6000 }
									fleet_event = { id = wg_crisis_2.401 days = 6001 }
								}

						}
					}

							OR = {
								has_global_flag = wg_crisis_very_easy
								has_global_flag = wg_crisis_easy
							}

					else_if = {

						country_event = { id = wg_crisis_2.14 days = 1080 }
					}
					else_if = {
						limit = { has_global_flag = wg_crisis_hard }
						country_event = { id = wg_crisis_2.14 days = 900 }
					}
					else = {
						country_event = { id = wg_crisis_2.14 days = 720 }
					}
					country_event = { id = wg_crisis_2.2000 days = 5 }
				}
				if = {
					# 简单难度开局刷2队弱的进攻舰队,1队弱的守家舰队
					limit = {
						OR = {
							has_global_flag = wg_crisis_very_easy
							has_global_flag = wg_crisis_easy
						}
					}
					create_wg_crisis_2_weaker_fleet_and_admiral = yes
				}
				if = {
					# 普通难度开局刷4队弱的进攻舰队,1队弱的守家舰队
					limit = { has_global_flag = wg_crisis_easy }
					create_wg_crisis_2_weaker_fleet_and_admiral = yes
				}
				
				if = {
					# 困难难度开局刷6队突袭舰队,1队守家舰队

						OR = {
							has_global_flag = wg_crisis_normal
							has_global_flag = wg_crisis_hard
							has_global_flag = wg_crisis_insane
						}
					}
					while = { count = 2
						create_wg_crisis_2_raider_fleet = yes
					}
					
					PREV = { star = {
						create_wg_crisis_2_guard_fleet = yes 
						last_created_fleet = { set_fleet_flag = first_portal_2_guard_fleet }
					}}
				}
				if = {
					# 萌新难度开局刷8队突袭舰队,2队守家舰队
					limit = {
						OR = {
							has_global_flag = wg_crisis_hard
							has_global_flag = wg_crisis_insane
						}
					}
					
					PREV = { random_system_planet = {
						limit = { is_primary_star = yes }
						create_wg_crisis_2_guard_fleet = yes	
					}}	

					while = { count = 2
						create_wg_crisis_2_raider_fleet = yes
					}
				}
				if = {
					# 天灾刷联合荣耀
					limit = { has_global_flag = wg_crisis_insane }
					create_wg_crisis_2_legendary_admiral = yes
					PREV = { star = { create_uf_glory_2 = yes }}	
				}
			}
			if = {
				limit = { exists = starbase }
				starbase = { fleet = { destroy_fleet = this } }
			}
			if = {
				limit = { NOT = { has_global_flag = wg_crisis_very_easy }}
				create_starbase = {
					size = wg_crisis_starbase
					owner = event_target:wg_crisis_2_country
					effect = { while = { count = 5 create_wg_crisis_platform = yes } }
				}
			}
		}
		every_playable_country = {
			limit = { 
				is_ai = no 
				has_wg_special_authority = yes
			}
			country_event = { id = wsg.3061 days = 540 random = 360 }
		}
		every_country = {
			limit = {
				is_default_and_fallen_country = yes
				intel_level = {
					level = high
					system = event_target:first_portal_2_system
				}
			}
			country_event = { id = wg_crisis_2.5 }
		}
		every_country = {
			limit = {
				is_default_and_fallen_country = yes
				NOT = {
					intel_level = {
						level = high
						system = event_target:first_portal_2_system
					}
				}
			}
			country_event = { id = wg_crisis_2.6 }
		}
		if = {
			limit = {
				has_global_flag = wg_crisis_doubles
				NOT = { has_global_flag = wg_invasion_happened }
			}
			country_event = { id = wg_crisis.3 }
		}
		set_update_modifiers_batch = end
	}
}

# Portal Destroyed (HIDDEN)
country_event = {
	id = wg_crisis_2.11
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = wg_crisis_disabled }
		NOT = { has_global_flag = wg_invasion_2_end }
		has_country_flag = wg_crisis_2_country
		FROMFROM = { OR = {
			is_ship_size = wg_dimensional_portal
			has_ship_flag = wg_portal_2
			Fleet = { has_fleet_flag = first_dimensional_portal_2 }
			Fleet = { has_fleet_flag = second_dimensional_portal_2 }
			Fleet = { has_fleet_flag = third_dimensional_portal_2 }
		}}
	}

	immediate = {
		if = {
			# 非天灾难度
			limit = {
				NOT = { has_global_flag = wg_crisis_insane }
			}
			set_global_flag = first_portal_2_destroyed
			FROM = { 
				save_event_target_as = portal_killer
				country_event = { id = wg_crisis_2.101 }
			}
			every_country = {
				limit = { has_point_of_interest = { poi = wg_invasion_2_poi.1 }}
				remove_point_of_interest = wg_invasion_2_poi.1
			}
		}
        else = {
			FROM = {
				save_event_target_as = portal_killer
			}
			FROMFROM.FLEET = {
				if = {
					# 第一传送门被破坏
					limit = {
						has_fleet_flag = first_dimensional_portal_2
					}
					set_global_flag = first_portal_2_destroyed
					every_country = {
						limit = { has_point_of_interest = { poi = wg_invasion_2_poi.1 }}
						remove_point_of_interest = wg_invasion_2_poi.1
					}
					if = {
						# 第二传送门未出现则马上刷出
						limit = {
							NOT = { has_global_flag = second_portal_2_opened }
						}
						event_target:wg_crisis_2_country = { country_event = { id = wg_crisis_2.15 }}
					}
					if = {
						# 所有传送门被破坏
						limit = {
							has_global_flag = second_portal_2_destroyed
							has_global_flag = third_portal_2_destroyed
						}
						event_target:first_portal_2_system = {
							save_event_target_as = last_portal_2_system
						}
						event_target:portal_killer = {
							country_event = { id = wg_crisis_2.101 }
						}
					}
					else = {
						# 尚有其他传送门
						event_target:first_portal_2_system = {
							save_event_target_as = destroyed_portal_2_system
						}
						every_playable_country = {
							country_event = { id = wg_crisis_2.53 }
						}
					}
				}
				else_if = {
					# 第二传送门被破坏
					limit = {

		set_variable = { which = num_flagship_exist value = 0 }

			limit = { 
				NOR = {  
					is_ship_size = scout_UAV

					AND = {
						fleet = { has_fleet_flag = counting_data }

							is_ship_size = uf_Juggernaut 
							is_ship_size = UF_Flagship
						}
					}
					NOT = { fleet = { has_fleet_flag = counting_data }}
				}
				fleet = { NOT = { has_fleet_flag = drones }}

				trigger = is_ship_size

				wg_battleship = { fleet = { change_variable = { which = num_bb_exist value = 1 }}}
				wg_battlecrusier = { fleet = { change_variable = { which = num_bc_exist value = 1 }}}
