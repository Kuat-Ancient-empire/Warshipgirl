country_event = {
	id = wg_astesia.1
	title = wg_astesia.1.title
	desc = wg_astesia.1.desc
	picture = GFX_evt_think_tank
	is_triggered_only = yes
	fire_only_once = yes
	
	trigger = {
		NOT = { has_country_flag = get_wg_astesia }
	}
	immediate = {
		
		if = {
			limit = { NOT = { any_country = { has_country_flag = get_wg_astesia }}}
			create_species = {
				name = Terra_Ultra
				plural = Terra_Ultra
				class = "Warship_Girls"
				portrait = wsg11
				name_list = WSG
				effect = {
					save_global_event_target_as = wg_astesia_species
				}
			}
			create_country = {
				name = "Name_leader_wg_astesia"
				species = event_target:wg_astesia_species
				type = wg_astesia
				flag = {
					icon = {
						category = "wg_event_flags"
						file = "wge_flags_10.dds"
					}
					background={
						category="backgrounds"
						file="stripes.dds"
					}
					colors={
						"blue"
						"indigo"
						"null"
						"null"
					}
				}
				effect = {
					save_global_event_target_as = wg_astesia_country
					set_country_flag = wg_astesia_country	
				}
			}
		}
		event_target:wg_astesia_country = {
			establish_communications_no_message = root
		}
		save_global_event_target_as = wg_astesia_own_country
		set_variable = { which = wg_astesia_affection value = 1 }
		set_variable = { which = wg_astesia_affection_level value = 1 }
		set_variable = { which = wg_astesia_level value = 10 }
		set_variable = { which = wg_astesia_cloth value = 0 }
		set_country_flag = get_wg_astesia
		country_event = {
			id = wg_astesia.2 days = 540
		}
		country_event = {
			id = wg_astesia.100 days = 360
		}
	}

	option = {
		name = wg_astesia.1.a
	}
}
