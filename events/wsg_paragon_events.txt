#不惧技能
planet_event = {
	id = wg_paragon.1019
	hide_window = yes
	is_triggered_only = yes

	# Prevent duplicate events
	trigger = {
		not = { has_planet_flag = checking_wg_impavido }
	}

	immediate = {
		set_planet_flag = checking_wg_impavido # Prevent duplicate events
		planet_event = {
			id = wg_paragon.1020
			days = 10
		}
	}
}

planet_event = {
	id = wg_paragon.1020
	hide_window = yes
	is_triggered_only = yes

	immediate = {
		if = {
			limit = { has_ground_combat = no }
			remove_planet_flag = checking_wg_impavido
		}
		else = {
			# Handle both sides of the ground combat
			random_kill_army_if_IMPAVIDO = { SIDE_A = attacker SIDE_B = defender }
			random_kill_army_if_IMPAVIDO = { SIDE_A = defender SIDE_B = attacker }
	
			# Do the same check again in X days
			planet_event = { id = wg_paragon.1020 days = 10 } # NOTE: Update custom loc for effect when changing days
		}
	}
}

country_event = {
	id = wg_paragon.1021
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		if = {
			limit = { NOT = { has_country_flag = wg_paragon.1005 } }
			random = {
				chance = 50
				set_country_flag = wg_paragon.1005
				country_event = { id = wg_paragon.1005 days = 360 random = 360  }#随机拿女灶神
			}
		}
		if = {
			limit = { NOT = { has_country_flag = wg_paragon.1022 } }
			random = {
				chance = 50
				set_country_flag = wg_paragon.1022
				country_event = { id = wg_paragon.1022 days = 360 random = 360  }#随机拿不惧
			}
		}
		if = {
			limit = { NOT = { has_country_flag = wg_paragon.1008 } }
			random = {
				chance = 50
				set_country_flag = wg_paragon.1008
				country_event = { id = wg_paragon.1008 days = 360 random = 360  }#随机拿U-1405
			}
		}
		if = {
			limit = { NOT = { has_country_flag = wg_paragon.1023 } }
			random = {
				chance = 30
				set_country_flag = wg_paragon.1023
				country_event = { id = wg_paragon.1023 days = 360 random = 360  }#随机拿北卡
			}
		}
		if = {
			limit = {
				NOT = { has_country_flag = wg_paragon.1024 }
				end_game_years_passed >= 0
			}
			random = {
				chance = 80
				set_country_flag = wg_paragon.1024
				country_event = { id = wg_paragon.1024 days = 360 random = 360  }#后期随机拿蒙大拿
			}
		}
		if = {
			limit = {
				NOT = { has_country_flag = wg_paragon.1025 }
			}
			random = {
				chance = 30
				set_country_flag = wg_paragon.1025
				country_event = { id = wg_paragon.1025 days = 360 random = 360  }#后期随机拿蒙大拿
			}
		}
		if = {
			limit = { NOT = { has_country_flag = wg_paragon.1011 } }
			random = {
				chance = 40
				set_country_flag = wg_paragon.1011
				country_event = { id = wg_paragon.1011 days = 360 random = 360  }#随机拿可畏
			}
		}
		if = {
			limit = { is_ai = no }
			country_event = { id = wg_paragon.1021 days = 7200 }
		}
	}
}
