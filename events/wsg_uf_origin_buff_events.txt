namespace = uf_buff

# 主触发，判断舰队里的船是否有某部件
country_event = {
	id = uf_buff.1
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		# 关掉所有buff的触发
		NOT = { has_global_flag = uf_buff_disabled }
		FROMFROM = { 
			OR = {
				any_owned_ship = {
					OR = {
						has_component = UF_AURA_C
						has_component = UF_AURA_D
						has_component = UF_AURA_E
						has_component = UF_AURA_F
						has_component = UF_AURA_Glory
					}
				}
				AND = { exists = leader leader = { has_leader_flag = uf_vv_admiral } }
				AND = { exists = leader leader = { has_leader_flag = wg_a150_admiral } }
				AND = { exists = leader leader = { has_leader_flag = wg_montana_admiral } }
				AND = { exists = leader leader = { has_leader_flag = sh_Vector_admiral } }
			}
            NOT = { has_fleet_flag = wsg_battle_insitution_cd }
        }
	}

	immediate = {
		fromfrom = {
		    if = {
				limit = { 
					any_owned_ship = {
						has_component = UF_AURA_C
					}
					NOT = { has_modifier = uf_buff_C }
				}
				add_modifier = { modifier = uf_buff_C days = 10 }	
			}			
		    if = {
				limit = { 
					any_owned_ship = {
						has_component = UF_AURA_D
					}
					NOT = { has_modifier = uf_buff_D }
				}
				add_modifier = { modifier = uf_buff_D days = 10 }	
			}
			if = {
				limit = { 
					any_owned_ship = {
						has_component = UF_AURA_E
					}
					NOT = { has_modifier = wg_timeline_invisable }
				}
				add_many_hp = { DAYS = 10 }
			}
			if = {
				limit = { 
					any_owned_ship = {
						has_component = UF_AURA_F
					}
					ROOT.fromfromfrom = { NOT = {
						has_modifier = uf_debuff_F
					} }
				}
				ROOT.fromfromfrom = { add_modifier = { modifier = uf_debuff_F days = 10 } }
			}
			if = {
				limit = { 
					any_owned_ship = {
						has_component = UF_AURA_Glory
					}
					NOT = { has_modifier = wg_timeline_invisable }
				}
				add_many_hp = { DAYS = 30 }
			}
			if = {
				limit = {
					exists = leader
					leader = { has_leader_flag = uf_vv_admiral }
					NOT = { has_modifier = wg_timeline_invisable }
				}
				add_many_hp = { DAYS = 5 }
			}
			if = {#超大和
				limit = {
					exists = leader
					leader = { has_leader_flag = wg_a150_admiral }
					NOT = { has_modifier = wg_a150_modifier }
				}
				add_modifier = { modifier = wg_a150_modifier days = 5 }
			}
			if = {#蒙大拿
				limit = {
					exists = leader
					leader = { OR = {
						has_leader_flag = wg_montana_admiral
						has_leader_flag = sh_Vector_admiral
					} }
					NOT = { has_modifier = wg_timeline_invisable }
				}
				add_many_hp = { DAYS = 3 }
			}
			if = {#Vector
				limit = {
					exists = leader
					leader = { has_leader_flag = sh_Vector_admiral }
					NOT = { has_modifier = sh_Vector_modifier }
				}
				add_modifier = { modifier = sh_Vector_modifier days = 30 }
			}
			set_timed_fleet_flag = { flag = wsg_battle_insitution_cd days = 30 }
		}
	}
}
