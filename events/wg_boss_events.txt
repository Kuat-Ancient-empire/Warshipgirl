							}
						}	
					}
					last_created_country = {
						set_country_flag = fairy_dragon_country
						save_global_event_target_as = fairy_dragon_country
						create_fleet = {
							name = "NAME_fairy_dragon2"

country_event = {
	id = wg_boss.125
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = yes

	trigger = {
		NOT = { has_global_flag = spawn_psi_systems }
	}

	immediate = {
		set_global_flag = spawn_psi_systems
		if = { # 生成在5跳外的星系
			limit = {
				any_system = {
					space_owner = { is_same_value = ROOT }
					distance = {
						source = ROOT.capital_scope
						type = hyperlane
						use_bypasses = no
						min_jumps = 5
					}
					any_system_planet = {
						has_owner = no
					}
				}
			}
			random_system_within_border = {
				limit = {
					distance = {
						source = ROOT.capital_scope
						type = hyperlane
						use_bypasses = no
						min_jumps = 5
					}
					any_system_planet = {
						has_owner = no
					}
				}
				save_global_event_target_as = wg_psionic_system_home
			}
		} else_if = { # 备用方案（找不到5跳之外的星系）
			limit = { any_system = {
				space_owner = { is_same_value = ROOT }
			}}
			random_system_within_border = {
				save_global_event_target_as = wg_psionic_system_home
			}
		} else_if = { # 太虚
			limit = { always = yes }
			random_controlled_fleet = {
				solar_system = {
					save_global_event_target_as = wg_psionic_system_home
				}
			}
		}

		event_target:wg_psionic_system_home = {
			set_star_flag = wg_psionic_system_home
			random_system_planet = {
				limit = { is_primary_star = yes }
				save_event_target_as = wg_psionic_system_home_star
			}
			spawn_megastructure = {
				type = wg_psionic_wormhole
				orbit_distance = 250
				orbit_angle = 0
				coords_from = event_target:wg_psionic_system_home_star
				init_effect = {
					create_ambient_object = {
						type = "wg_psi_object"
						location = this
					}
					last_created_ambient_object = { 
						save_global_event_target_as = wg_psionic_wormhole_target
						set_ambient_object_flag = wsg_psi_object_flag
						set_location = prev
					}
				}	
			}
		}
		set_spawn_system_batch = begin
		no_scope = {
			spawn_system = {
				min_distance >= 602
				max_distance <= 606
				min_orientation_angle = 224
				max_orientation_angle = 226
				initializer = "wg_psi_1_system"
				is_discovered = no
				hyperlane = no
			}
			event_target:wg_psi_1_system = {
				spawn_system = {
					min_distance >= 20
					max_distance >= 30
					min_orientation_angle = 200
					max_orientation_angle = 206
					initializer = "wg_psi_2_system"
					hyperlane = no
					is_discovered = no
				}
				spawn_system = {
					min_distance >= 20
					max_distance >= 30
					min_orientation_angle = -50
					max_orientation_angle = -45
					initializer = "wg_psi_3_system"
					hyperlane = no
					is_discovered = no
				}
			}
		}
		set_spawn_system_batch = end
		event_target:wg_psi_1_system = {
			every_fleet_in_system = {
				fleet_event = { id = wg_boss.134 }
			}
			add_hyperlane = {
				from = event_target:wg_psi_1_system
				to = event_target:wg_psi_2_system
			}
		}
		event_target:wg_psi_2_system = {
			every_fleet_in_system = {
				fleet_event = { id = wg_boss.134 }
			}
			add_hyperlane = {
				from = event_target:wg_psi_2_system
				to = event_target:wg_psi_3_system
			}
		}
		event_target:wg_psi_3_system = {
			every_fleet_in_system = {
				fleet_event = { id = wg_boss.134 }
			}
			add_hyperlane = {
				from = event_target:wg_psi_1_system
				to = event_target:wg_psi_3_system
			}
		}
		country_event = { id = wg_boss.136 days = 5 }
	}
}

# 多次进入星系总触发
# 使用此触发的星系必须要带 wg_event_system 这个flag，用来先进行筛选
ship_event = {
	id = wg_boss.10002
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		owner = { is_ai = no }
		# 初步筛选条件
		FROM = { has_star_flag = wg_event_system }
	}

	immediate = {
		if = {
			limit = { FROM = { has_star_flag = wg_boss_lly_system }}
			if = {
				limit = { exists = event_target:wg_boss_lly_fleet }
				root.owner = { 
					if = {
						limit = { 
							has_country_flag = establish_communicate_with_lly 
							NOR = { 
								has_country_flag = give_food_lly 
								has_country_flag = lly_ask_for_food
								has_country_flag = lly_ask_food_cooldown
							}
						}
						set_country_flag = lly_ask_for_food
						country_event = { id = wg_boss.226 }
					}
					else_if = {
						limit = { NOT = { has_country_flag = encountered_wg_boss_lly }}
						set_country_flag = encountered_wg_boss_lly
						country_event = { id = wg_boss.209 }
					}
				}
			}
		}
	}
