country_event = {
	id = uf_origin.8
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		has_authority = auth_united_fleet
	}

	immediate = {
		owner_main_species = {
			modify_species = {
				species = this
				base = none
				remove_trait = uf_trait_united_fleet
				add_trait = uf_trait_united_fleet_1
				effect = {
					save_event_target_as = uf_species 
				}
			}
		}
		event_target:uf_species = {
			set_species_identity = owner_main_species
		}
		change_dominant_species = { species = event_target:uf_species change_all = yes }
		every_owned_leader = {
			limit = { 
				OR = { 
					has_trait = leader_trait_uf 
					is_same_species = event_target:uf_species
				}
			}
			remove_trait = leader_trait_uf
			add_trait_no_notify = leader_trait_uf_1
			if = {
				limit = { leader_class = scientist }
				remove_trait = leader_trait_scientist_uf
				add_trait_no_notify = leader_trait_scientist_uf_1
			}
		}
		every_pool_leader = {
			limit = {
				OR = {
					is_same_species = root
				}
				NOR = {
					has_trait = leader_trait_uf_1
				}
			}
			add_trait_no_notify = leader_trait_uf_1
			remove_trait = leader_trait_uf
			if = {
				limit = { leader_class = scientist }
				remove_trait = leader_trait_scientist_uf
				add_trait_no_notify = leader_trait_scientist_uf_1
			}
		}
		every_owned_leader = {
			limit = {
				has_leader_flag = uf_vv_admiral
				NOT = { has_trait = leader_trait_admiral_uf_vv_4 }
			}
			change_leader_portrait = wsg112
			switch = {
				trigger = leader_class
				admiral = { 
				    remove_trait = leader_trait_admiral_uf_vv
					#remove_trait = leader_trait_admiral_uf
					add_trait_no_notify = leader_trait_admiral_uf_vv_1 
					#add_trait_no_notify = leader_trait_admiral_uf_1 
				}
			}
		}
		every_owned_leader = {
			limit = {
				has_leader_flag = uf_Yubari_scientist
			}
			change_leader_portrait = wsg9983
			switch = {
				trigger = leader_class
				scientist = { 
				    remove_trait = leader_trait_scientist_uf_Yubari
					add_trait_no_notify = leader_trait_scientist_uf_Yubari_2
				}
			}
		}
		country_event = { id = wsg.9997 days = 1 }
	}
}

planet_event = {
	id = uf_origin.10001
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = {
			has_planet_flag = uf_planet_no_assimilation_flag
		}
		any_owned_pop = { species = {
			NOT = {
				is_same_species = root.owner_main_species
			}
		}}
	}

	immediate = {
		# 判断异族人口是否少于6个
		if = {
			limit = { count_owned_pop = {
				limit = { species = {
					NOT = { is_same_species = root.owner_main_species }
				}}
				count <= 6
			}}
			while = {
				limit = { any_owned_pop = { species = {
					NOT = { is_same_species = root.owner_main_species }
				}}}
				random_owned_pop = {
					limit = {
						NOT = { is_same_species = root.owner_main_species }
					}
					kill_pop = yes
				}
				create_pop = {
					species = owner_main_species
				}
			}
		} else = {
			random_list = {
				1 = { # 6 pops
					while = {
						count = 6
						random_owned_pop = {
							limit = { species = {
								NOT = { is_same_species = root.owner_main_species }
							}}
							kill_pop = yes
						}
						create_pop = {
							species = owner_main_species
						}
					}
				}

				1 = { # 12 pops
					while = {
						count = 12
						random_owned_pop = {
							limit = { species = {
								NOT = { is_same_species = owner.owner_main_species }
							}}
							kill_pop = yes
						}
						create_pop = {
							species = owner_main_species
						}
					}
				}
				1 = { # 20 pops
					while = {
