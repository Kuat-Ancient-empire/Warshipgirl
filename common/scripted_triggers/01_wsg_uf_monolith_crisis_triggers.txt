# scope = system
monolith_target_system_trigger = {
	NOR = {
		has_star_flag = monolith_crisis_shield_system
		has_star_flag = monolith_crisis_systems
	}
	OR = {
		NOT = { exists = space_owner }
		AND = {
			exists = space_owner
			space_owner = { NOR = { 
				is_same_value = event_target:$OWNER$ 
				is_same_value = event_target:$OWNER_2$ 
			}}
		}
		any_system_planet = { 
			has_owner = yes 
			owner = { NOT = { is_country_type = primitive }}
			count_fleet_in_orbit = { count = 0 limit = {
				exists = owner
				owner = { OR = { is_same_value = event_target:$OWNER$ is_same_value = event_target:$OWNER_2$ } }
				NOT = { is_same_value = root }
			} }
		}
	}
}
# scope = planet
monolith_target_planet_trigger = {
	has_owner = yes 
	owner = { NOT = { is_country_type = primitive }}
	count_fleet_in_orbit = { count = 0 limit = {
		exists = owner
		owner = { OR = { is_same_value = event_target:$OWNER$ is_same_value = event_target:$OWNER_2$ } }
		NOT = { is_same_value = root }
	} }	
}
