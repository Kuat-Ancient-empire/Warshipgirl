	user_scope = fleet
	button_visible = {
		always = yes
	}
	button_clickable = {
		custom_tooltip = {
			is_in_combat = no
			exists = solar_system
			solar_system = {
				OR = {
					has_global_flag = ignore_uf_portal_enemy_fleet_trigger
					NOT = {
						any_fleet_in_system = {
							exists = owner
							owner = {
								is_hostile = prevprev.from
							}
						}
					}
				}
				NOT = {
					any_system_megastructure = {
						OR = {
							is_megastructure_type = uf_mobile_portal
							is_megastructure_type = uf_mobile_portal_temp
							is_megastructure_type = gateway_final
							is_megastructure_type = gateway_restored
						}
					}
				}
			}
		}
	}
	wait = no
	on_completed = on_wg_uf_mobile_gateway_activated
	tooltip = wg_uf_mobile_gateway_button
