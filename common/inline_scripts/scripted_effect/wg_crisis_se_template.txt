create_wg_crisis$CURRENT$_colossus = {
	create_fleet = {
		name = "NAME_wg_crisis_colossus_fleet"
		effect = {
			set_owner = event_target:wg_crisis$CURRENT$_country
			set_location = {
				target = PREV
				distance = 60
				angle = random
			}
			create_ship = {
				name = random
				design = "NAME_UF_Star_eater"
				graphical_culture = "wsg_01"
				upgradable = no
			}
			set_fleet_stance = passive
			set_fleet_flag = colossus_fleet
			event_target:wg_crisis$CURRENT$_country = {
				change_variable = {
					which = uf_fleet_id
					value = 1
				}
			}
			event_target:wg_crisis$CURRENT$_country = {
				change_variable = {
					which = uf_colossus_id
					value = 1
				}
			}
			solar_system = {
				set_variable = {
					which = fleet_id
					value = prev.uf_fleet_id
				}
			}
			set_variable = {
				which = "uf_fleet_id"
				value = event_target:wg_crisis$CURRENT$_country.uf_fleet_id
			}
			set_variable = {
				which = uf_colossus_id
				value = event_target:wg_crisis$CURRENT$_country.uf_colossus_id
			}
			save_event_target_as = uf_colossus_fleet
			fleet_event = { id = wg_crisis.97 }
		}
		settings = {
			can_upgrade = no
			uses_naval_capacity = no
			can_change_leader = no
			can_change_composition = no
			can_disband = no
			spawn_debris = no
			ai_ignore_strength = yes
		}
	}
}

spawn_fleets_insane$CURRENT$ = {
	optimize_memory
    while = { count = 3
        create_wg_crisis$CURRENT$_raider_fleet = yes
    }
    while = { count = 1
        fleet_event = { id = wg_crisis$CURRENT$.4 days = 200 }
    }
    fleet_event = { id = wg_crisis$CURRENT$.4 days = 360 } 
    fleet_event = { id = wg_crisis$CURRENT$.4 days = 540 } 
    fleet_event = { id = wg_crisis$CURRENT$.4 days = 720 }
    while = { count = 1
        create_wg_crisis$CURRENT$_greater_fleet = yes
        last_created_fleet = {
            set_name = "NAME_auxiliary_guard_fleet"
            set_timed_fleet_flag = { flag = defence_fleet days = 4500 }
            set_formation_scale = 2
            fleet_event = { id = wg_crisis$CURRENT$.401 days = 4501 }
        }
    }
    while = { count = 1
        create_wg_crisis$CURRENT$_greater_fleet = yes
        last_created_fleet = {
            set_name = "NAME_auxiliary_guard_fleet"
            set_timed_fleet_flag = { flag = defence_fleet days = 6000 }
            set_formation_scale = 2
            fleet_event = { id = wg_crisis$CURRENT$.401 days = 6001 }
        }
    }
    fleet_event = { id = wg_crisis$CURRENT$.102 days = 9000 } #循环刷辅助守家舰队
}
