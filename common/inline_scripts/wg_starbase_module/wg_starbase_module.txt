		is_wsg_arkship = yes
	}
	resources = {
		category = starbase_arkship_upgrades
		cost = {
			alloys = @arkship_cost_alloys_100
		}
		upkeep = {
			energy = @arkship_upkeep_energy_2
		}
	}
	cannot_fit_on_slots = {
		spinal_mount_1
	}
	station_modifier = {
		show_only_custom_tooltip = no
		megastructure_build_speed_mult = 0.25
	}
	planet_modifier = {
		planet_building_capacity_add = 100
	}
	on_finished = {
		set_ship_construction_type = starbase_shipyard
	}
	on_destroyed = {
		if = {
			limit = {
				NOT = {
					has_starbase_module = wsg_arkship_engineering_bay
				}
			}
			set_ship_construction_type = none
		}
	}
	
	ai_weight = {
		weight = 100		#use same weight here - ai weighting is done in starbase_types.txt
	}
}

wsg_arkship_grand_archive = {
	icon = "GFX_arkship_grand_archive"
	construction_days = 1800
	starbase_type = arkship
	category = sm_utility_cat
	empire_limit = {
		base = 1
	}
	potential = {
		is_wsg_arkship = yes
	}
	cannot_fit_on_slots = {
		spinal_mount_1
	}
	resources = {
		category = starbase_arkship_upgrades
		cost = {
			alloys = @arkship_cost_alloys_500
			volatile_motes = @arkship_cost_strategic_100
		}
		upkeep = {
			energy = @arkship_upkeep_energy_10
		}
	}
	country_modifier = {
		country_vivarium_max_capacity_add = @grand_archive_vivarium_capacity
	}
	on_finished = {
		owner = {
			if = {
				limit = {
					has_country_flag = grand_archive_destroyed
				}
				remove_country_flag = grand_archive_destroyed
			}
			set_country_flag = has_arkship_grand_archive_module
			country_event = {
				id = grand_archive.4120
			}
			country_event = {
				id = grand_archive.4199
			}
			if = {
				limit = {
					has_origin = origin_treasure_hunters
				}
				country_event = {
					id = grand_archive.8090
					scopes = {
						from = prev.solar_system
					}
				}
			}
		}
	}
	on_destroyed = {
		owner = {
			remove_country_flag = ga_destroyed_contains_space_fauna
			if = {
				limit = {
					num_vivarium_slots > 0
				}
				set_country_flag = ga_destroyed_contains_space_fauna
				release_all_vivarium_fauna = {
					LOCATION = prev.solar_system
				}
			}
			set_country_flag = grand_archive_destroyed
			remove_country_flag = has_arkship_grand_archive_module
		}
	}
	ai_weight = {
		weight = 100		#use same weight here - ai weighting is done in starbase_types.txt
	}
}
