
# 参数如下：
# 本次危机编号 : CURRENT 
# 相同房间 : $ROOM2$ $ROOM3$
# 不同房间 : $ROOM$ $ROOM1$
# 样本effect : $WG_END_SPECIMAN_EFFECT$ 
# 移除额外兴趣点 : $WG_END_REMOVE_INTEREST_EFFECT$ 
# 海宅刷新舰队 : $WG_INITIAL_SPECIAL_EFFECT$ 
# END GAME PRICE DESIGN : $WG_PRICE_DESIGN_NAME$ 
# END GAME PRICE FLEET NAME : $WG_PRICE_FLEET_NAME$

country_event = {
	id = wg_crisis$CURRENT$.3
	hide_window = yes
	fire_only_once = yes
	is_triggered_only = yes

	trigger = {
		NOR = { 
			has_global_flag = wg_crisis_disabled
			has_global_flag = wg_invasion$CURRENT$_happened
			has_global_flag = wg_invasion_end
		}
	}

	immediate = {
		set_update_modifiers_batch = begin
		if = {
			limit = { has_global_flag = activated_uf_hidden_diff }
			remove_global_flag = wg_crisis_very_easy
			remove_global_flag = wg_crisis_easy
			remove_global_flag = wg_crisis_normal
			remove_global_flag = wg_crisis_hard
			set_global_flag = wg_crisis_insane
		}
		endgame_telemetry = united_fleet
		set_global_flag = wg_invasion$CURRENT$_happened
		initialize_crisis_system_select = yes
		country_event = { id = wg_crisis.21 }
		country_event = { id = wg_crisis.62 days = 360 }
		set_timed_global_flag = { flag = resistance_reinforce_cooldown days = 359 }
		set_global_flag = resistant_army_will_defence
		country_event = { id = wg_crisis.31 days = 20 random = 30 }
		every_playable_country = {
			limit = { is_ai = no }
			country_event = { id = wg_crisis.75 days = 20 }
		}
		random_system = {
			limit = {
				NOR = { 
					has_star_flag = forbit_spawn_system
					is_star_class = sc_black_hole 
					any_system_planet = {
						is_planet_class = pc_black_hole
					}
				}
			}
			set_star_class = sc_wsg_uf
			set_star_flag = first_portal$CURRENT$_system
			set_star_flag = cannot_be_hit_by_aow_starfall_spear
			save_global_event_target_as = first_portal$CURRENT$_system
			random_system_planet = {
				if = {
					limit = { NOT = { has_global_flag = wsg_any_player_civic_wsg_super_harbor } }
					initial_uf_crisis_country = { THIS = wg_crisis$CURRENT$ }
				}
				else = { initial_uf_crisis_country = { 
					THIS = wg_crisis$CURRENT$ 
					UF_COUNTRY_NAME = NAME_WG_harbor_crisis_country
				} }
				event_target:wg_crisis$CURRENT$_country = {
					create_fleet = {
						name = "NAME_Dimensional_Portal"
						effect = {
							set_owner = event_target:wg_crisis$CURRENT$_country
							create_portal = yes
							set_location = {
								target = PREVPREV
								distance = 40
								angle = random
							}
							save_global_event_target_as = first_dimensional_portal$CURRENT$$CURRENT$
							set_fleet_flag = first_dimensional_portal$CURRENT$
							set_event_locked = yes
							#基础2个普通入侵舰队
							fleet_event = { id = wg_crisis$CURRENT$.4 days = 180 } 
							fleet_event = { id = wg_crisis$CURRENT$.4 days = 360 } 

							if = {
								# 普通以上难度开门后额外刷2队舰队
								limit = {
									OR = {
										has_global_flag = wg_crisis_normal
										has_global_flag = wg_crisis_hard
										has_global_flag = wg_crisis_insane
									}
								}
								fleet_event = { id = wg_crisis$CURRENT$.4 days = 540 } 

								create_wg_crisis$CURRENT$_greater_fleet = yes
								last_created_fleet = {
									set_name = "NAME_auxiliary_guard_fleet"
									set_timed_fleet_flag = { flag = defence_fleet days = 4500 }
									set_formation_scale = 2
									fleet_event = { id = wg_crisis$CURRENT$.401 days = 4501 }
								}
							}
							
							if = {
								# 困难以上难度开门后额外刷2队舰队
								limit = {
									OR = {
										has_global_flag = wg_crisis_hard
										has_global_flag = wg_crisis_insane
									}
								}
								fleet_event = { id = wg_crisis$CURRENT$.4 days = 720 } 
								fleet_event = { id = wg_crisis$CURRENT$.4 days = 900 } 
								fleet_event = { id = wg_crisis$CURRENT$.102 days = 10800 } #循环刷辅助守家舰队
							}

							if = {
								# 天灾难度开门后再刷1队
								limit = { has_global_flag = wg_crisis_insane }

								fleet_event = { id = wg_crisis$CURRENT$.4 days = 1080 } 
								fleet_event = { id = wg_crisis$CURRENT$.4 days = 1260 } 

								create_wg_crisis$CURRENT$_greater_fleet = yes
								last_created_fleet = {
									set_name = "NAME_auxiliary_guard_fleet"
									set_timed_fleet_flag = { flag = defence_fleet days = 6000 }
									fleet_event = { id = wg_crisis$CURRENT$.401 days = 6001 }
								}
							}
							
						}
					}
					if = {
						limit = { # 循环刷舰队事件
							OR = {
								has_global_flag = wg_crisis_very_easy
								has_global_flag = wg_crisis_easy
							}
						}
						country_event = { id = wg_crisis$CURRENT$.14 days = 1260 }
					}
					else_if = {
						limit = { has_global_flag = wg_crisis_normal }
						country_event = { id = wg_crisis$CURRENT$.14 days = 1080 }
					}
					else_if = {
						limit = { has_global_flag = wg_crisis_hard }
						country_event = { id = wg_crisis$CURRENT$.14 days = 900 }
					}
					else = {
						country_event = { id = wg_crisis$CURRENT$.14 days = 720 }
					}
					$WG_INITIAL_SPECIAL_EFFECT$
				}
				if = {
					# 简单难度开局刷1队弱的进攻舰队
					limit = {
						OR = {
							has_global_flag = wg_crisis_very_easy
							has_global_flag = wg_crisis_easy
						}
					}
					create_wg_crisis$CURRENT$_weaker_fleet_and_admiral = yes
				}
				if = {
					# 普通难度开局刷1队弱的进攻舰队
					limit = { has_global_flag = wg_crisis_easy }
					create_wg_crisis$CURRENT$_weaker_fleet_and_admiral = yes
				}
				if = {
					# 困难难度开局刷2队突袭舰队,1队守家舰队
					limit = {
						OR = {
							has_global_flag = wg_crisis_normal
							has_global_flag = wg_crisis_hard
							has_global_flag = wg_crisis_insane
						}
					}
					while = { 
						count = 2
						create_wg_crisis$CURRENT$_raider_fleet = yes
					}
					
					PREV = { star = {
						create_wg_crisis$CURRENT$_guard_fleet = yes 
						last_created_fleet = { set_fleet_flag = first_portal$CURRENT$_guard_fleet }
					}}
				}
				if = {
					# 萌新难度开局刷2队突袭舰队,1队守家舰队
					limit = {
						OR = {
							has_global_flag = wg_crisis_hard
							has_global_flag = wg_crisis_insane
						}
					}	

					while = { 
						count = 2
						create_wg_crisis$CURRENT$_raider_fleet = yes
					}

					PREV = { random_system_planet = {
						limit = { is_primary_star = yes }
						create_wg_crisis$CURRENT$_guard_fleet = yes	
					}}
				}
				if = {
					# 天灾刷联合荣耀
					limit = { has_global_flag = wg_crisis_insane }
					create_wg_crisis_legendary_admiral = { THIS = wg_crisis$CURRENT$ }
					PREV = { star = { create_uf_glory$CURRENT$ = yes }}	
				}
			}

			if = {
				limit = { exists = starbase }
				starbase = { fleet = { destroy_fleet = this } }
			}

			if = {
				limit = { NOT = { has_global_flag = wg_crisis_very_easy }}
				create_starbase = {
					size = wg_crisis_starbase
					owner = event_target:wg_crisis$CURRENT$_country
					effect = { while = { count = 5 create_wg_crisis_platform = yes } }
				}
			}
		}
		every_playable_country = {
			limit = { 
				is_ai = no 
				has_wg_special_authority = yes
			}
			country_event = { id = wsg.3061 days = 540 random = 360 }
		}
		every_country = {
			limit = {
				is_default_and_fallen_country = yes
				intel_level = {
					level = high
					system = event_target:first_portal$CURRENT$_system
				}
			}
			country_event = { id = wg_crisis$CURRENT$.5 }
		}
		every_country = {
			limit = {
				is_default_and_fallen_country = yes
				NOT = {
					intel_level = {
						level = high
						system = event_target:first_portal$CURRENT$_system
					}
				}
			}
			country_event = { id = wg_crisis$CURRENT$.6 }
		}
		set_update_modifiers_batch = end
	}
}

# Portal Destroyed (HIDDEN)
country_event = {
	id = wg_crisis$CURRENT$.11
	hide_window = yes
	is_triggered_only = yes

	trigger = {
		NOT = { has_global_flag = wg_crisis_disabled }
		NOT = { has_global_flag = wg_invasion_end }
		has_country_flag = wg_crisis$CURRENT$_country
		FROMFROM = { OR = {
			is_ship_size = wg_dimensional_portal 
			has_ship_flag = wg_portal$CURRENT$
			Fleet = { has_fleet_flag = first_dimensional_portal$CURRENT$ }
			Fleet = { has_fleet_flag = second_dimensional_portal$CURRENT$ }
			Fleet = { has_fleet_flag = third_dimensional_portal$CURRENT$ }
		}}
	}

	immediate = {
		if = {
			# 非天灾难度
			limit = {
				NOT = { has_global_flag = wg_crisis_insane }
			}
			set_global_flag = first_portal$CURRENT$_destroyed
			FROM = { 
				save_event_target_as = portal_killer
				country_event = { id = wg_crisis$CURRENT$.101 }
			}
			every_country = {
				limit = { has_point_of_interest = { poi = wg_invasion$CURRENT$_poi.1 }}
				remove_point_of_interest = wg_invasion$CURRENT$_poi.1
			}
		}
        else = {
			FROM = {
				save_event_target_as = portal_killer
			}
			FROMFROM.FLEET = {
				if = {
					# 第一传送门被破坏
					limit = {
						has_fleet_flag = first_dimensional_portal$CURRENT$
					}
					set_global_flag = first_portal$CURRENT$_destroyed
					every_country = {
						limit = { has_point_of_interest = { poi = wg_invasion$CURRENT$_poi.1 }}
						remove_point_of_interest = wg_invasion$CURRENT$_poi.1
					}
					if = {
						# 第二传送门未出现则马上刷出
						limit = {
							NOT = { has_global_flag = second_portal$CURRENT$_opened }
						}
						event_target:wg_crisis$CURRENT$_country = { country_event = { id = wg_crisis$CURRENT$.15 }}
					}
					if = {
						# 所有传送门被破坏
						limit = {
							has_global_flag = second_portal$CURRENT$_destroyed
							has_global_flag = third_portal$CURRENT$_destroyed
						}
						event_target:first_portal$CURRENT$_system = {
							save_event_target_as = last_portal$CURRENT$_system
						}
						event_target:portal_killer = {
							country_event = { id = wg_crisis$CURRENT$.101 }
						}
					}
