	category = "planet_automation_wg_building_space"
	available = {
		free_amenities <= -5
		owner = {
			has_wg_authority = no
			NOR = {
				has_authority = auth_machine_intelligence
				has_authority = auth_hive_mind
			}
		}
		OR = {
			NOT = { uses_district_set = city_world }
			free_district_slots = 0

			}

		commerce = {
			building = building_commercial_zone
			available = {
				owner = {
					is_megacorp = yes
				}
			}
		}
	}
}
