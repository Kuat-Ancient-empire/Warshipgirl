#深海建造器
building_sh_assembly_plant = {
	base_buildtime = 240
	planet_limit = 1

	category = pop_assembly
	
	potential = {
		exists = owner
		owner = { 
			has_authority = auth_shenhai	
			NOT = { has_ascension_perk = ap_become_the_gray }
		}
	}

	building_sets = {
		government
		urban
	}

	convert_to = {
		"building_sh_nano_assembly_plant"
	}
	custom_tooltip = wg_planet_pop_assembly_average_int_tooltip_1
	triggered_planet_modifier = {
		bonus_pop_growth = 1
		multiplier = planet.value:wg_planet_pop_assembly_average_int
	}
	resources = {
		category = planet_buildings
		cost = {
			minerals = 250
		}
		upkeep = {
			minerals = 2
			energy = 2
		}
		upkeep = {
			trigger = {
				exists = assembling_species
				assembling_species = {
					OR = {
						has_trait = sh_trait_building2E
						has_trait = sh_trait_building22E
					}
				}
			}
			sr_se = 5
		}
	}	
	
	prerequisites = {
		"tech_sh_assembly"
	}
	
	upgrades = {
		building_sh_assembly_plantU
	}
}
