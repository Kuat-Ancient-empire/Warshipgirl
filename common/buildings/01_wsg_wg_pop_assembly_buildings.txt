building_warshipgirl_assembly_plant = {
	base_buildtime = 240
	planet_limit = 1
	category = pop_assembly
	potential = {
		exists = owner
		owner = {
			is_regular_empire = yes
			has_authority = auth_warshipgirls
			NOR = {
				has_ascension_perk = ap_project_s
				has_ascension_perk = ap_become_the_gray
			}
		}
	}
	building_sets = {
		government
		urban
	}
	convert_to = {
		"building_warshipgirl_senergy_assembly_plant"
		"building_warshipgirl_nano_assembly_plant"
	}
	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				has_wg_special_authority = no
				has_ascension_perk = ap_project_s
				has_ascension_perk = ap_become_the_gray
			}
		}
	}
	custom_tooltip = wg_planet_pop_assembly_average_int_tooltip_1
	triggered_planet_modifier = {
		bonus_pop_growth = 1
		multiplier = planet.value:wg_planet_pop_assembly_average_int
	}	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 300
		}
		upkeep = {
			minerals = 5
			energy = 5
		}
	}
	prerequisites = {
		"tech_warshipgirl_assembly_plant"
	}
	upgrades = {
		building_warshipgirl_assembly_plantU
	}
}
