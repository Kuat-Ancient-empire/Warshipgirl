building_sh_nerve = {
	base_buildtime = 180
	base_cap_amount = 1

	category = unity
	building_sets = {
		government
		unity
	}
	potential = {
		exists = owner
		OR = {
			owner = { is_ai = no }
			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
			has_authority = auth_shenhai	
		}
		NOT = { has_modifier = slave_colony }
	}
	
	destroy_trigger = {
		exists = owner
		OR = {
			owner = { is_regular_empire = no }
			AND = {
				owner = { 
					is_ai = yes 
					buildings_no_sr_pantsu = yes
				}
				free_jobs = 0
				OR = {
					free_building_slots = 0
					is_planet_class = pc_habitat
				}
			}
		}
	}
	
	planet_modifier = {
		planet_jobs_produces_mult = 0.20
		planet_amenities_add = -1000
	}
	
	resources = {
		category = planet_buildings
		cost = {
			minerals = 500
		}
		upkeep = {
			sr_pantsu = 2
			energy = 8
		}
	}	
	
	prerequisites = {
		"tech_sh_nerve"
	}	
	ai_resource_production = {
		minerals = 20
		energy = 20
		food = 20
		trigger = { always = yes }
	}
}

			free_building_slots > 0
		}
		owner = { 
			is_regular_empire = yes
		}
		NOT = { has_modifier = slave_colony }
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = 800
		}
		upkeep = {
			energy = 10
		}
	}
	
	planet_modifier = {
		planet_technician_energy_produces_add = 1
		planet_miners_minerals_produces_add = 1
		planet_farmers_food_produces_add = 1
		district_generator_max = 2
		district_mining_max = 2
		district_farming_max = 2
		planet_max_districts_add = 1
	}
	#轨道
	triggered_planet_modifier = {
		potential = { uses_district_set = habitat }
		district_generator_max = -2
		district_mining_max = -2
		district_farming_max = -2
	}
	triggered_planet_modifier = {
		potential = { 
			OR = {
				colony_type = col_generator 
				colony_type = col_habitat_energy 
			}	
		}
		planet_technician_energy_produces_add = 1
	}
	triggered_planet_modifier = {
		potential = { 
			OR = {
				colony_type = col_mining 
				colony_type = col_habitat_mining
			}	
		}
		planet_miners_minerals_produces_add = 1
	}
	triggered_planet_modifier = {
		potential = { colony_type = col_farming }
		planet_farmers_food_produces_add = 1
	}
	triggered_desc = {
		text = sh_col_notice
	}
	
	prerequisites = {
		"tech_building_sh_base_resources_department"
	}
	upgrades = {
		building_sh_base_resources_department_2
	}
}
