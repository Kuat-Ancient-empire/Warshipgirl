		if = {
			limit = {
				exists = event_target:wg_fallen_empire

				NOR = {
					has_ascension_perk = ap_shenhai_silent

					has_civic = civic_wsg_super_harbor

				}
			}

				country_event = {
					id = wg_astesia.128

			}
		}
		else = {
			prev = {
				country_event = {
					id = wg_astesia.127
				}
			}
		}
	}
}
